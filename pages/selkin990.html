ئەخمەتجان قۇربان سابىرىنىڭ ئۇيغۇر بالىلار مەسەللىرى ناملىق كىتاۋى ھەققىدە .

بۇ يازما matimatika تەرپىدىن يورۇتۇلدى(2007-10-10)
رەسىم:
رەسىم:
ئەخمەتجان قۇربان سابىرى 1960-يىلى9-ئايداقورغاس ناھىيىسىنىڭ كۆكتالا دېھقانچىلىق مەيدانىدا بىر ھەربىي ئائىلىسىدە دۇنياغا كەلگەن، 1978-يىلى شىنجاڭ ئۇنۋېرسىتىتىنىڭ ئەدەبىيات فاكولتىتىنغا كىرىپ، 1983-يىلى ئوقۇش پىتتىرىپ،شىنجاڭ تېلېۋېزىيە ئستانسىسىغا خىزمەتكە كىرگەن.
ئۇ ئۇيغۇر تېلېۋىزىيە سەنئەتىنىڭ دەسلەپكى شەكىللىنىشى ئىچىن تۆھبە قوشقان.ھەمدە 300دىن ئارتۇق قەدىمىي ۋەھازىرقى زامان خەنزۇچە شېئىرلارنى تەرجىمە قىلىپ، تېلېۋىزىيە پروگراممىلاردىكى بوشلۇقنى تولدۇردى. ئۇ،1994-يىلىدىن تارتىپ، "يىكسەل قورغاس"،"سېھىرلىك سىمپونىيە"،"پايتەختتىكى تەڭرىتاغلىقلار"،"باغرىڭنى ئاچ باغراش"،"خىسلەتلىك بۇلاق" قاتارلىق500دىن ئارتۇق تېلېۋىزىيە سەنئەت فىلىمىنىڭ سىنارىيىسى،"ياشلىق ساداسى"،"نۇرۇز ناۋاسى"،"ناۋالىق دىيار --تۇرپانيىزى"،"ناۋا ۋە جۇلا"،"دۇنيا ۋە ئاياللار"،"جېلىليىزدە مەشرەپ "قاتارلىق 50سەنئەت كېچىلىكىنىڭ سىنارىيىسىنى يازغان. ئۇنىڭ "قىممەتلىك ھايات"،"يوقلاپ تۇرگىن ئاتا-ئاناڭنى"،"سىڭلىمغا"،"كارۋان ھېكمىتى"،"جۇپ پەرزەنت"،"نۇرۇز ناۋاسى"،"ھايات دېگەن سىرلىق تېپىشماق"قاتارلىق150دىن ئوشۇق ناخشا تېكىستى رادىئو-تېلېۋىزىيدە ئاڭلىتىلماقتا. ئۇ يازغان "قىز بالا بولسا نېمە بوپتۇ؟"،باجگىرنىڭ باش قېتىمچىلىقى"،"بالىلىرىنى كىتىش"،"ئانىسىز ئۆي "،"مېھرى شەپقەت"،"يىراق يېزىدىكى مۇھەببەت"قاتارلىق 50دىن ئارتۇق ئىتوتى تېلېۋىزۇرلاردا كۆرسىتىلمەكتە.ئۇ يەنە ھەرخىل ژانىردىكى 500دىن ئارتۇق شېئىر؛"چاقىرغۇ ئاسقان قىزلار"،"يالغۇز تام "،"بىزنىڭ دۇكان"،"يېڭى باشلىق كېلىدۇ"،"تىڭىرقىغان ئەپسانىلەر"،"سەھەردە كۆرگەن چىش"قاتارلىق يۇمۇرستىك پوۋېست،ھېكايە،"يىللار ۋە ئىنسانلار"،"شېئىر موھىت،شېئىر كەڭلىك،شېئىر تىل ھەققىدە ئويلىغانلىرىم"،"ھېكايىدىكى شېئىر ۋە موزىكا"،"ئىلى خەلق چاقچاقلىرى ھەققىدە دەسلەپكى مۇلاھىزە"قاتارلىق 100دىن ئارتۇق ئەدەبىي ئوبزۇر ۋە ئىلمىي ماقالىنى ئېلان قىلغان.
ئۇنىڭ" بالىلار يۇمۇرلىرى"،"بالىلار چاقچاقلىرى"، "كىن دە بىر كىلكە"، "بالىلار تېپىشماقلىرى"(5 قىسىم)،"كالتە قۇيرۇق چاشقان"(چۆچەكلەر توپلىمى)،تونۇر بېشىدىكى چۆچەكلەر"(3قىسىم)، ئۇيغۇر بالىلار تېپىشماقلىرى"(6قىسىم)،"ئۇيغۇر بالىلار قوشاقلىرى" (5قىسىم) قاتارلىق كىتاپلىرى ئوقۇرمەنلەر بىلەن يىز كۆرىشتى. "ياشلىق ئىزنالىرى"، "قاينامدىكى غەۋۋاس"، "بالىلارغا باغلانغان قەلپ"، "ھەممە ژانىرغا قول تىققان ئەدىب"دېگەن ماۋزۇدا< ئىن-سىن گىلزارى>، <شىنجاڭ ياشلىرى>، <ئىلى دەرياسى>، <تەڭرىتاغ>قاتارلىق ژورناللار ئۇنى تونۇشتۇردى، ئۇ يازغان"كۆك ئۆستەڭ"ناملىق يۇمۇرستىك رومانى ،."ئۇيغۇر بالىلار مەسەللىرى"دېگەن 5 قىسىملىق كىتابى نەشىردىن چىقىپ ، ئوقۇرمەنلەر بىلەن يىز كۆرىشتى  "ئەلقۇتنىڭ يۇرتىنى ساياھەت قىلىشى"، "نەسىردىن ئەپەندىم ھەققىدە يېڭى قىسسە" ناملىق ھەجۋىلىك رومانى يېقىندا ئوقۇرمەنلەر بىلەن يىز كۆرىشىدۇ.

مەسەللەر خۇددى چۆچەكلەرگە ئوخشاش ، بالىلارمۇ، چوڭلارمۇ ئورتاق ياقتۇرىدىغان ئەدەبىي ژانىرلارنىڭ بىرى ، ئۇ چۆچەككە قارىغاندا تېخىمۇ ئىخچام ، جەلىپكار بولۇپ ، ئۆزىگە تېخىمۇ مول پەلسەپىۋى پىكىرلەرنى يۇشۇرغان بولىدۇ ، كۆپلىگەن مەسەللەر بالىلىرىمىزنىڭ ئۆگىنىش ، تۇرمۇش ۋە ئۆسۈپ يىتىلىش مۇھىتىغا ماسلاشتۇرۇلغان ، بالىلىرىمىزنىڭ  پىسخىكىلىق ئالاھىدىلىكى بۇيىچە بايان قىلىنغان .  

يېقىندا ئەخمەتجان قۇربان سابىرى بالىلار ئۈچۈن نەشىر قىلدۇرغان «ئۇيغۇر بالىلار مەسەللىرى (1-5) » ئۇيغۇر مەسەللىرى خەزىنىسىگە قۇشۇلغان يېڭى يۇلتۇز، ئۇيغۇر بالىلىرى ئۈچۈن قىلىنغان ئەڭ قىممەتلىك سوۋغا بولۇپ ھىساپلىنىدۇ . بۇ جەمئىي بەش پارچە كىتاب بولۇپ ،   
1 .   كالتە قۇيرۇقلۇق چاشقان .
2. بالىلار يۇمۇرلىرى .
3. بالىلار چاخچاقلىرى .
4.تۇنۇر بېشىدىكى چۆچەكلەر( 3 قىسىم  )
5 . ئۇيغۇرچە - خەنزۇچە بالىلار تېپىشماقلىرى ( 5 قىسىم )

يوقۇرىقىسى ئەخمەتجان قۇربان سابىرى ۋە ئۇنىڭ مەسەللەر توپلىمىغا مۇناسىۋەتلىك بىر قىسىم سۈرەتلەر. (داۋاملىق قۇشۇلىدۇ . )

سەلكىن تور بېكىتى ۋە مۇنبىرىدىكى بارلىق ئەزالار ۋە زىيارەتچىلەر دىققىتىگە :

تور بېكىتىمىزدە ، جۈملىدىن مۇنازىرە مۇنبىرىمىزدە دۆلەت قانۇنىغا خىلاپ بولغان، مىللەتلەر ئىتتىپاقلىقىغا تەسىر كۆرسىتىدىغان ، دۆلەت سىياسىتىگە قارشى مەزمۇندىكى،  شەخس ياكى كوللىكتىپنىڭ مەنپەئەتىگە بىۋاستە بۇزغۇنچىلىق قىلىش ، ھوجۇم قىلىش ، ھاقارەتلەش خاراكتىرىدىكى  تىما ۋە ئىنكاسلارنىڭ بولۇشى قەتئىي چەكلىنىدۇ، خىلاپلىق قىلغۇچىلار( يوللىغۇچى ياكى تەستىقلىغان باشقۇرغۇچى ) مۇناسىۋەتلىك ئورۇنلارنىڭ بىر تەرەپ قىلىشىغا تاپشۇرۇلىدۇ، مۇنبەردە بەدىئىي زوق بىرىدىغان ، بىلىم يەتكۈزۈش ، ئۇچۇر ئالماشتۇرۇش ۋە كۆڭۈل ئېچىش خاراكتىرىدىكى مەزمۇنلار ئاساس قىلىنىشى  كىرەك، بارلىق ئەزالار ۋە باشقۇرغۇچىلارنىڭ قاتتىق دىققەت قىلىشىنى ئالاھىدە ئەسكەرتىمىز!!! مۇنبەردە يوقارقىدەك مەزمۇنلارنى بايقىغانلارنىڭ مۇنبەر باشلىقى ياكى باشقۇرغۇچىلىرىغا ئەسكەرتىپ قويۇشىنى ئۈمىت قىلىمىز!!!   ( سەلكىن تورى )
