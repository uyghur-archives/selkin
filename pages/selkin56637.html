 ساۋاتسىز قالغان باي 1مىليون 400مىڭ يۈەن مەبلەغ چىقىرىپ ، يۇرتىغا مەكتەپ سالدى.

باشقۇرۇش ئەسكەرىتمىسى :
بۇ يازمىنى ئالتۇن دۇنياغا نەزەر دىن بۇ سەھىپىگە يۆتكەپ كەلدى(2009-02-02)
ئابلىكىم كامالنىڭ ئاتا-ئانىسى بىر قېتىملىق ئاھالە كۆچۈرۈشتە قاراقاش ناھىيسنىڭ پۇرچاقچى يېزىسىدىن ناھيىنىڭ ئەڭ چېتى بولغان تۇزلۇقئوتاق رايونىغا بارغانىدى. بۇ چۆلدە مەكتەپ بولمىغاچقا ئابلىكىم ساۋاتسىز قالدى .37ياشقا كىرگەن ئابلىكىم كامال ھازىرمۇ بانكىغا پۇل قويۇپ -ئالغاندا ئىسمىنى يازغاندىن باشقا ھېچقانداق خەت يازالمايدۇ.ساۋاتسىزلىق ئۇنىڭ قەلبىنى تولىمۇ ئازابلىغاچقا ، ئۇ، كۈنلەرنىڭ بىرىدە مەكتەپ سېلىشنى ئۆز غايسى قىلىپ ياشىدى .
ئابلىكم كامال1985-يىلى تۇزلۇقئوتاق رايونىدىن ئۆزى باققان 38تۇياق قوينى ئېلىپ پۇرچاقچى يېزىسىغا قايتىپ چىقاندىن كېيىن بورداقچىلىق ،دان ئايرىش ماشىنىسى سېتىۋىلىپ يۇرتداشلىرىنىڭ خامىنىنى ئېلىپ بېرىش ، توك تۈگمىنى سېتىۋىلىپ ئۇن تارتىش بىلەن ھاللىنىپ روناق تاپتى ؛ ئۈرۈمچى ۋە ئىچكى ئۆلكىلەرگە بېرىپ قاسساپلىق ،كاۋاپچىلىق ، ئاشخانا ئېچىش بىلەن تىجارىتىنى يەنىمۇ كېڭەيتىپ ،قاراقاشتىكى ساناقلىق بايلارنىڭ بىرىگە ئايلاندى . بۇ جەرياندا ئۇ دوستلىرى بىلەن بىرلىشىپ ئىچكى ئۆلكىلەر ۋە ئوتتۇرا ئاسىيا دۆلەتلىرىگە بېرىپ تىجارەت قىلىپ ،ئۆلكە ،دۆلەت ھالقىپ تىجارەت قىلىشنىڭ قانۇنىيتنى ئوبدان ئىگىلۋالدى .ئەمما خەنزۇ تىلى ۋە چەت ئەل تىللىرىنى پەقەت بىلمىگەچكە ،ساۋاتسىزلىق ۋە تىل بىلمەسلىكنىڭ دەردىنى يەتكۈچە تارتتى ؛نۇرغۇن پۇلى تەرجىمانلارنىڭ چۆنتىكىگە چۈشۈپ كەتتى .ئۇ ئويلاندى : ئۆز يېزىقىمنى ، خەنزۇ تىل - يېزىقىنى،چەتئەل تىللىرىنى بىلگەن بولسام ،تېخىمۇ كۆپ پۇل تاپار ئىدىم ، خەپ!
ئۆزۇم چوڭ بولغان يۇرتتا بىر باشلانغۇچ مەكتەپ سالدۇرۇپ ، بىرمۇ ئۆسمۈرنىڭ مەندەك ساۋاتسىز قېلىپ ، پۇشايمان يېيشىگە يول قويمايمەن ، دېگەن ئىرادىگە كەلدى .دېمىسىمۇ ،2003-يىلى ئۇ. ئۆز يېزىسىدىكى 160ئائىلىگە 400مىڭ يۈەنگە يېقىن ئۆسۈمسىز قەرز بېرىپ.ئۇلارنىڭ تېزرەك نامراتلىقتىن قۇتۇلشىنى ئۈمىد قىلغانىدى .ئەمسۇس ،ئۇلارنىڭ كۆپىنچىسى ساۋاتسىز بولغاچقا ئىشنىڭ ئېپىنى بىلمىگەنلىكتىن ،پۇلنى جايىدا ئىشلىتەلمەي ،ئىككى يىل ئىچىدىمۇ نامراتلىقتىن قۇتۇلالمىدى.بۇ ئاچچىق ساۋاق ئابلىكىم كامالغا ساۋاتسىزلىقنىڭ ئۆزىگىمۇ،ئۆزگىگىمۇ مۆلچەر- لىگۈسىز زور زىيان ئەكىلدىغانلىقىنى يەنە بىر قېتىم تونۇتتى .
ئابلىكىم كامالنىڭ ئارزۇسى ئەمەلگە ئاشىدىغان پەيىت ئاخىرى يېتىپ كەلدى.
ئۇ 1مىليون 400مىڭ يۈەن مەبلەغ سالغان ، 460نەپەر ئوقۇغۇچى ئوقۇيالايدىغان «پۇرچاقچى يېزا ئىگىلىك تىكلەش ئۈمىد باشلانغۇچ مەكتىپى»گە ئۇل سېلىش مۇراسىمى بۇ يىل 29-ئاپرېلدا يۇرتى پۇرچاقچىدا تەنتەنلىك ئۆتكۈزۈلدى .
خوتەن ۋىلايەتلىك پارتىكوم ۋە مەمۇرىي مەھكىمىدىن ئىككى نەپەر رەھبەر ئۇل سېلىش مۇراسىمىغا قاتنىشىپ ،سېخى -مەردانە دېھقان ئوغلانى ئابلىكىم كامالنىڭ مەرىپەت سۆيەر روھىغا يۇقىرى باھا بەردى ۋە ئالدىن بېيىغان باشقا كىشلەرنىڭمۇ ئۇنىڭ روھىدىن ئۈگۈنۈپ ، مائارىپقا ياردەم بىرشىنى ئۈمىد قىلدى . ئابلىكىم كامال مۇنداق دېدى :«مەن مەكتەپ پۈتكەندىن كېيىن ئوقۇتۇش ئەسلىھەلىرىنى تولۇقلاپ بېرىش بىلەن بىرگە ، ئىچكى ئۆلكىلەردىن ئىلغار ئوقۇتۇش ئۈسكۈنىلىرىنى ، تەجرىبە سايمانلىرىنى مەكتەپكە سەپلەپ بېرىپ مەكتەپنى قاراقاش ناھيىسى بويىچە ئالدىنقى قاتاردىكى مەكتەپ قىلىپ قۇرۇپ چىقىمەن دېدى»

مۇشۇ ئادەم توغۇرلۇق مۇنداق بىر پاراڭ بولغان ئىدى..
مەن بۇ ئادەمنىڭ رەسىمىنى شىنجاڭ مەدەنىيتىنىڭ مەلۇم بىر سانىنىڭ مۇقاۋىسىدا كۆرگەندىن كېيىن ئاندىن ئىش ئىزلىرىنى ئوقۇشقا قىززىقىپ قالدىم..ئۇقوپ بولغاندىن كېيىن بۇ ئادەمنىڭ قارىقاش ناھيسىدىن ئىكەنلىكىنى بىلدىم دە قارىقاشلىق بىر دوستۇمغا ۋاي قارا...سىنىڭ يۇرتۇڭدىن مۇشۇنداق سىخى ئادەم چىقىپتۇ دىسەم ئۇ دوستۇم قوپۇپ...بۇنداق قىمارۋازلىق قىلىپ پۇل تىپىپ بۇنداق ئىش قىلغۇچە غىت قىسىپ جىم يۈرگەن ياخشى دىدى....بۇ گەپنى ئاڭلاپ شۇنداق ئاچچىقىم كەلدىكى..دوستۇمغا (ئۇ دوستۇمنىڭ نەچچە مىليۇن پۇلى بار ئىدى )سەن تىخى ھاجىم.. شۇنداق تۇرۇقلۇق باشقىلارغا قىلچىلىك ياخشىلىق قىلمايسەن قاچانلا بولسا ماشىناڭدا غىلجىڭ خىنىملار كەم بولمايدۇ،يامان يولغا خەجلىسەڭ خەجلىدىڭكى ياخشى ئىشلار ئۈچۈن پۇل سەرپ قىلدىمۇ دىسەم ، ئۇ دوستۇم تۇرۇپ ئىشقىلىپ مەن پۇلنى ئۇنىڭدەك ھارامدىن تاپمىدىم دەپلا گەپنى ئۈزدى.
